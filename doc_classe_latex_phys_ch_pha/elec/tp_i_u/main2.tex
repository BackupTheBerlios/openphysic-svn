\tp[Propriétés des intensités et des tensions]{Propriétés des intensités \\et des tensions}

\nomprenomclasse

\objectifs{
\item Réaliser un schéma de montage où figurent les appareils de mesure (ampèremètre, voltmètre) et leurs bornes
\item \'Etude des propriétés de l'intensité d'un courant dans un circuit électrique
\item \'Etude des propriétés de la tension dans un circuit électrique
}


\materiel{
\item générateur $6$-$12~V$ réglable
\item 3 multimètres
\item 2 lampes de tension nominale $6~V$
\item $R = 10~\Omega$ ; $R_1 = 180~\Omega$ ; $R_2 = 270~\Omega$
\item 1 bouton poussoir
}
