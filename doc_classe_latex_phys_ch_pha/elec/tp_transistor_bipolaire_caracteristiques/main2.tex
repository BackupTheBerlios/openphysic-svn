\Section{Montage Emetteur Commun}

}



% montage


\begin{itemize}
\item La jonction base émetteur est polarisée en direct.
\item La jonction collecteur base est polarisée en inverse.
\end{itemize}


\Section{Caractéristique d'entrée : $I_B = f(V_{BE})$ pour $V_{CE} = cste$}

\begin{enumerate}
\item \`A l'aide de $E_B$, on fait varier $I_B$ de $0$ à environ $110~\mu A$ et on relève les valeurs de $V_{BE}$, $I_B$ et $I_C$ en maintenant $V_{CE} = 5~V$.

\item Tracer la caractéristique $I_B = f(V_{BE})$ et en déduite la résistance dynamique $\displaystyle r = \frac{\Delta V_{BE}}{\Delta I_B}$

\item Interpréter les résultats.

\end{enumerate}



\Section{Caractéristique de transfert de courant : $I_C = f(I_B)$ pour $V_{CE} = cste$}

\begin{enumerate}
\item Tracer la caractéristique $I_C = f(I_B)$ pour $V_{CE} = 5~V$.

\item Calculer le facteur d'amplification en courant : $\displaystyle \beta = \frac{\Delta I_C}{\Delta I_B}$.

\item Interpréter les résultats.
\end{enumerate}



\Section{Caractéristique de sortie : $I_C = f(V_{CE})$ pour $I_B = cste$}
\begin{enumerate}
\item On maintient $I_B$ constant. On fait varier $V_{CE}$ et on relève $I_C$ et $V_{CE}$.

 
\item Faire le tableau des mesures pour $I_B = 20\ 40\ 60\ 80\ 100~\mu A$


% tableau


\item Tracer les caractéristiques $I_C = f(V_{CE})$ et calculer, dans la partie rectiligne de la première et dernière courbe, la résistance $\displaystyle R = \frac{\Delta V_{CE}}{\Delta I_C}$


\item Interpréter les résultats.

\end{enumerate}
