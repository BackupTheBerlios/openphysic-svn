\tp{\'Etude d'un circuit $R$-$L$-$C$ en régime sinusoïdal}

\objectifs{
\item tracer l'évolution 
        \begin{itemize}
        \item de l'intensité efficace $I$ du courant
        \item de l'impédance $Z$
        \item du déphasage $\varphi$
        \end{itemize}
en fonction de la fréquence $f$ pour un circuit $R$-$L$-$C$ série
alimenté par une tension sinusoïdale.
}


\bigskip


%\Section*{Travail expérimental}

\begin{enumerate}

\item Régler, avec le GBF, $u(t)$ : signal sinusoïdal avec $\hat{U} =
2,5~V$ et $f = 100~Hz$.

\vressort{1}

\item Câbler le montage ci-dessous en plaçant les connections
permettant de visualiser $u(t)$ et $u_R(t)$ en même temps. \emph{faire
vérifier le montage par le professeur avant la mise sous tension}.


% fig
\vspace{3cm}


\begin{center}
$C = 1~\mu F$ (boîte de capacité $\times~1\mu F$

$R = 1~k \Ohm$ (boîte de résistances $\times~1~k\Ohm$)

$L = 1~H$ (bobine à noyau réglable)
\end{center}



\item Représenter sur le même oscilligramme $u_R(t)$ et $u(t)$. En
déduire le déphasage $\varphi$ \emph{de $u_R(t)$ par rapport à $u(t)$}
(valeur absolue et signe) et l'impédance $Z$ du circuit.

\vressort{1}

\item Mesurer $U_L$ et $U_C$ avec un voltmètre en précisant la
position du sélecteur.

\vressort{1}

\item Tracer le diagramme de Fresnel du circuit qui vérifie la loi des
mailles en prenant $1~cm \equivalent 0,4~V$.\\Retrouver la valeur de
$\varphi$ mesurée au (3).

\vressort{1}

\item \emph{\'Eteindre le GBF} et insérer dans le montage un
ampèremètre permettant de mesurer $I$.\\
\emph{Faire vérifier par le professeur avant la mise sous tension.}

\vressort{1}

\item Faire varier $f$ de $50$ à $500~Hz$ et mesurer
\begin{itemize}
\item $\varphi$ (préciser à chaque mesure son signe et l'exprimer en
 $°$)
\item $Z$
\item $I$
\end{itemize}
On prendra une dizaine de valeurs de $f$ (5 points avant et 5 points
après la résonance).\\
Présenter les résultats sous forme de tableau.

\vressort{1}

\item Tracer les courbes :
\begin{itemize}
\item $I = I(f)$
\item $Z = Z(f)$
\item $\varphi = \varphi(f)$
\end{itemize}
Commenter ces 3 courbes.\\
En déduire la fréquence de résonance $f_0$ puis le coefficient de
surtension $Q$.\\
Comparer $f_0$ avec la valeur théorique et $Q$ avec des rapport des
tensions mesurées à $f = f_0$.

\vressort{1}

\item Quelle est l'influence sur les valeur de $I$ et $\varphi$ si $R$
passe de $1~k\Ohm$ à $5~k\Ohm$ ?

\vressort{1}

\item Passer l'oscilloscope en mode XY.\\
Faire varier $f$ de $50$ à $500~Hz$. Qu'observez-vous ?

\end{enumerate}
