\section{\'Etude}

Le système mécanique étudié est la masse $m$.
\begin{enumerate}
\item Faire le bilan des forces (extérieures) agissant sur la masse $m$.

\vressort{1}

\item Représenter les forces sur un schéma.

\vressort{1}

\item \'Enoncer la loi d'équilibre de la masse $m$. En déduire une relation liant l'intensité $P$ du poids et la tension $T$ du ressort.

\end{enumerate}


\vressort{2}


\section{Mesures}

\begin{enumerate}
\item Mesurer la longueur $l_0$ du ressort à vide.

\vressort{1}

\item Poser différentes masses $m$ à l'extrémité du ressort, mesurer la longueur du ressort en charge, déduire l'allongement $\Delta l = l - l_0$ de celui-ci.

\vressort{1}

\item Noter les résultats expérimentaux dans le tableau suivant :


\begin{arraydata}{9}
\hline
\rule[-0.4cm]{0cm}{1cm}%

\rule[-0.4cm]{0cm}{1cm}%

\rule[-0.4cm]{0cm}{1cm}%
$T$ ($N$)  & & & & & & & & & \\ \hline
\rule[-0.4cm]{0cm}{1cm}%
$\Delta l$ ($m$)  & & & & & & & & & \\ \hline
\end{arraydata}


\vressort{1}

\item Tracer la courbe $T = f(\Delta l)$.

\vressort{1}

\item Exploiter les paramètres de la courbe (Est-ce une droite ? Passe-t-elle par l'origine ? Quel est son coefficient directeur ? Quelle est l'unité de ce coefficient directeur ?).

\vressort{1}

\item En déduire la relation (vectorielle) liant la tension $\vect{T}$ du ressort avec l'allongement de celui-ci.

\end{enumerate}
