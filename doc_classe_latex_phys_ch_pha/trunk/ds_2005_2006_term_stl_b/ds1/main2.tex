\begin{exercice}
On associe une résistance $R_1$ en série avec deux résistances $R_2$ et $R_3$ associées en parallèle. On applique une tension $E$ à l'ensemble.


\begin{enumerate}
\item Faire le schéma du montage.
\item Déterminer l'intensité du courant traversant $R_1$.
\item Calculer la tension $U_{R_1}$ aux bornes de $R_1$.
\item Calculer la tension $U_{R_{23}}$ aux bornes de $R_2$ et $R_3$.
\item Calculer l'intensité du courant traversant $R_2$.
\item Calculer l'intensité du courant traversant $R_3$.
\end{enumerate}


\donnees{
\item $R_1 = 56~\Ohm$ ; $R_2 = 68~\Ohm$ ;  $R_3 = 82~\Ohm$ ; $E = 6~V$
}

\end{exercice}

\begin{exercice}
Un générateur réel ($E$, $r$) est associé en série avec deux résistances $R_1$ et $R_2$.

\begin{enumerate}
\item Comment s'appellent $E$ et $r$ ?
\item Faire le schéma du montage.
\item Calculer l'intensité du courant $I$ à l'aide de la loi de Pouillet (ou de la loi des mailles et la loi d'Ohm).
\end{enumerate}

\donnees{
\item $E = 12~V$ ; $r = 5~\Ohm$ ; $R_1 = 100~\Ohm$ ; $R_2 = 220~\Ohm$
}

\end{exercice}
