\begin{enumerate}
\item Rappelez le symbole d'une lentille mince convergente et celui d'une

trois rayons particuliers traversant la lentille.
\item Même question pour une lentille mince divergente.
\item Rappelez la formule de conjugaison des lentilles minces.
\item Rappelez les deux formules de grandissement.

\begin{exercice}{\\}
On place un objet lumineux plan $AB$ de $1~cm$ de hauteur, à $6~cm$ en
avant d'une lentille $L$ convergente de centre optique $O$ et de
distance focale $f'=4~cm$.

\begin{enumerate}
\item Déterminez la vergence de la lentille.
\item Faites un schéma à l'échelle $1$, trouvez la position de
  l'image $A'B'$ ainsi que sa taille $\ma{A'B'}$.
\item Donnez les caractéristiques de l'image $A'B'$ (réelle/virtuelle,
  même sens que l'objet/renversée, agrandie/rétrécie)
\item Déterminez par le calcul la position de l'image $A'B'$.
\item Déterminez par le calcul la grandeur de l'image $A'B'$.
\end{enumerate}
\end{exercice}


\vressort{1}

\begin{exercice}{\\}
Une lentille convergente a une distance focale $f' = 10~cm$. Un objet de $1~cm$ de haut se trouve à $5~cm$ en avant de la lentille.

\begin{enumerate}
\item Calculez la vergence de la lentille.
\item Notez sur un schéma à l'échelle $1$ ses foyers $F$ et $F'$.
\item Déterminez par construction la position, la nature et la taille
  de l'image.
\item  Déterminez par calcul la position, la nature et la taille
  de l'image.
\end{enumerate}

\end{exercice}
