% Interro optique - Lentilles minces


\renewcommand{\reproduire}{
\begin{enumerate}
\item Rappelez le symbole d'une lentille mince
  \begin{enumerate}
  \item convergente,
  \item divergente.
  \end{enumerate}
\item Placez les deux foyers $F$ et $F'$ et le centre optique $O$ dans le cas d'une lentille mince 
  \begin{enumerate}
  \item convergente,
  \item divergente.
  \end{enumerate}
\item Représentez les trois rayons particuliers traversant une
  lentille mince
  \begin{enumerate}
  \item convergente,
  \item divergente.
  \end{enumerate} 
\item Rappelez la formule de conjugaison des lentilles minces et
  précisez les unités.
\item Rappelez les deux formules de grandissement et précisez les
  unités.
\item Définissez la vergence d'une lentille et précisez \emph{les}
  unités.
\end{enumerate}
}
