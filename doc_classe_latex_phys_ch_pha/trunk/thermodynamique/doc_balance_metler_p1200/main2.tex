\begin{enumerate}
\item Vérifier l'horizontalité de la balance (vis de calage + bulle B

\item Allumer. La balance doit être à $000,00~g$. L'élève précedent
  doit l'y avoir mise (sinon l'y mettre). Le bouton de tarage doit
  être lui aussi à zéro ( en butée)

\item Si pesée directe d'un récipient R

  \begin{enumerate}
    \item Poser délicatement R sur le plateau

    \item Si échelle lumineuse dépassée (le signe + apparaît),
      abaisser délicatement les centaines de $g$ nécessaires.

    \item Régler les centièmes de $g$

    \item Relever  $m = \Box\Box\Box,\Box\Box~g$

   \end{enumerate}

  \item Si pesée d'une poudre ou d'un liquide :

   \begin{enumerate}
     \item Tarer le récipient R :
       \begin{itemize}
         \item le poser (vide) sur le plateau
         \item tarer à zéro (bouton TARA $0-100$ + molette de tarage
           fin)


           \emph{NE PLUS TOUCHER AU TARRAGE PENDANT LA PES\'EE}

       \end{itemize}

     \item Avec spatule verser les cristaux jusqu'à masse de
       poudre désirée (ou verser le liquide)

     \item Régler les centièmes de $g$.
       
     \item Relever $m = \Box\Box\Box,\Box\Box~g$

   \end{enumerate}

\item \'Eteindre la balance. Enlever le récipient du plateua. Rallumer
  et mettre tout à zéro (bouton de tarage ; centaines de $g$ ;
  centième de $g$)
   

\end{enumerate}
