\objectifs{
\item Déterminer $L_f$ la chaleur latente de fusion de la glace
}

\vressort{1}

\materiel{
\item glaçons
\item calorimètre et ses accessoires
\item thermomètre
\item bécher gradué
}


\vressort{3}

\section{Manipulation}
\begin{enumerate}
\item Sortir les glaçons du réfrigérateur. Ils fondent partiellement.

\item Introduire un masse $m_1$ d'eau dans le calorimètre à l'aide du
bécher gradué (environ $300~g$).

\item Noter, à l'équilibre thermique, la température initiale
$\theta_1$. Reporter les mesures dans le tableau final.

\item Introduire une masse $m_2$ (environ $50$ à $70~g$ de glaçons
secs à $\theta_2 = 0~°C$ (glace fondante). Reporter les mesures dans
le tableau final.

\item \`A l'équilibre thermique, noter la température finale
$\theta_3$.

\end{enumerate}

\vressort{3}

\section{Principe : calculs théoriques}

\begin{enumerate}
\item L'eau et le calorimètre cèdent une quantité de chaleur $Q_1$
telle que $Q_1 = m_1 c (\theta_3 - \theta_1) + C_{\mbox{\small calo}}
(\theta_3 - \theta_1)$

\item La glace fond et l'eau obtenue passe de $\theta_2 = 0~°C$ à
$\theta_3$ (température d'équilibre final). La quantité de chaleur
$Q_2$ reçue pour réaliser cette transformation peut se décomposer en
deux parties !

        \begin{enumerate}

                \item ${Q'}_2 = m_2 \cdot L_f$ : chaleur reçue par la glace
                pour sa fusion.

                \item ${Q"}_2 = m_2 \cdot c \cdot (\theta_3 -
                \theta_2) = m_2 \cdot c \cdot \theta_3 \mbox{ (car }
                \theta_2 = 0~°C \mbox{)}$ : chaleur reçue par l'eau de
                fusion de la glace pour s'échauffer jusqu'à
                $\theta_3$.

        \end{enumerate}

\item Dans l'enceinte adiabatique, on a l'équation calorimétrique $Q_1
+ Q_2 = 0$ c'est à dire $Q_1 + {Q'}_2 + {Q"}_2 = 0$.


Donc $m_1 \cdot c \cdot (\theta_3 - \theta_1)
 + C_{\mbox{\small calo}} \cdot (\theta_3 - \theta_1)
 + m_2 \cdot L_f
 + m_2 \cdot c \cdot \theta_3
 =0
$

Soit

\[
L_f = \troufixe{10cm}
\]

\end{enumerate}

\bigskip


\vressort{3}


Calculer $\Delta L_f = |{L_f}_{\mbox{\small th\'eorique}} -
{L_f}_{\mbox{\small exp\'erimental}}|$ : erreur absolue sur
l'expérience.

\medskip

Calculer $\frac{\Delta L_f}{{L_f}_{\mbox{\small
th\'eorique}}} \cdot 100$, qui donne le pourcentage d'erreur
relative.


On obtient ainsi une bonne estimation de la précision sur la
détermination expérimentale de $L_f$.


\vressort{3}


\[
\left\lbrace
\begin{array}{l}
\Delta L_f = \troufixe{3cm} \\
\frac{\Delta L_f}{{L_f}_{\mbox{\small th\'eorique}}} \cdot 100 = \troufixe{3cm}\\
\end{array}
\right.
\]
