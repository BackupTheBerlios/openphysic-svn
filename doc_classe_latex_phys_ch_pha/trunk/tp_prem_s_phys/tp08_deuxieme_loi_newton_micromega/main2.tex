\section{Protocole 1}
\subsection{Réalisation}

\begin{enumerate}

\item On étudie le mouvement d'un palet autoporteur sur un plan
  incliné, lancé initialement avec une vitesse non nulle. Dans tout le
  TP, les frottements seront négligés. Citer le référentiel
  d'étude. Faire le bilan des forces appliquées au palet une fois
  qu'il a été lancé. Compléter le schéma ci-dessous avec les vecteurs
  forces appropriés.


\item Lancer le logiciel Microméga 1ère S. Puis simulateurs. Puis
  mobile autoporteur, mouvement sans frottement sur un plan incliné.

\item Onglet affichage : cocher «~vue de face~», «~marquer la
  trajectoire~», «~afficher le repère~», «~afficher la grille~».

\item Onglet vecteurs : cocher «~afficher le vecteur vitesse~».

\item Onglet chronophotographie : cocher «~conserver les vecteurs
  représentés~», «~marquer uniquement les positions du centre
  d'inertie~». Puis activer la chronophotographie.

\item Imprimer autant de fois que d'élèves dans le groupe («~animation
  vue de face~»). Numéroter les points apparus de $0$ à $9$.

\end{enumerate}

\subsection{Exploitation}

\begin{enumerate}
\item Construire $\vect{\Delta V_1} = \vect{V(t_2)} - \vect{V(t_0)}$
  puis $\vect{\Delta V_6} = \vect{V(t_7)} - \vect{V(t_5)}$ à
  l'extérieur de la figure.

\item Reporter $\vect{\Delta V_1}$ au point $1$ puis $\vect{\Delta
    V_6}$ au point $6$.

\end{enumerate}

\section{Protocole 3}

\subsection{Réalisation}

\begin{enumerate}
\item Cliquer sur effacer en haut à gauche.

\item Onglet paramètres : $\beta = 20°$, $V_0 = 0~m.s^{-1}$, $X_0 =
  5~m$, $Y_0 = 10~m$. Le palet est lâché sans vitesse initiale.

\item Onglet affichage : Cocher "vue de face", "marquer la
  trajectoire", "afficher le repère", "afficher la grille".

\item Onglet vecteurs. Cocher "affichage vecteur vitesse", Choisir
  comme échelle de représentation : mini + 6 clics.

\item Onglet chromatographie. Cocher "conserver les vecteurs",
  "marquer les positions du centre d'inertie". Activer
  chronophotographie.

\item Imprimer autant de fois que d'élèves dans le groupe.
\end{enumerate}

\vressort{1}

\subsection{Exploitation}

\begin{enumerate}
\item Rappeler le bilan des forces appliquées au palet.
\item Construire de façon externe à la figure $\vect{\Delta V_5} = \vect{V(t_6)} - \vect{V(t_4)}$.
\item Conclure.
\end{enumerate}
