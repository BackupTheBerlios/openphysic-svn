\tp[Vérification expérimentale
de la deuxième loi de Newton dans
le cas d'une trajectoire parabolique]{Vérification expérimentale\\
de la deuxième loi de Newton dans\\
le cas d'une trajectoire parabolique
}


\objectifs{
\item Vérifier que le vecteur $\vect{\Delta V}$ est colinéaire et de même
  sens que la force appliquée à un solide.
}

\materiel{
\item caméra vidéo
\item téléviseur
\item magnétoscope
\item papier calque
}


\vressort{3}

\section{Protocole}

\begin{enumerate}

\item \`A l'aide d'une caméra vidéo, enregistrer le mouvement d'une balle de
golf lancée dans un plan frontal, c'est à dire dans un plan
perpendiculaire à l'axe de visée de la caméra.

\item \`A l'aide d'un magnétoscope, faire défiler l'enregistrement
  image par image et pointer le centre de la balle sur le transparent
  collé sur l'écran du téléviseur.

\end{enumerate}

\newpage

\section{Exploitation de l'enregistrement}
\`A partir du transparent, déterminer :

\indent
\begin{itemize}
\item la vitesse instantanée en chacune des positions enregistrées
\item le vecteur $\vect{\Delta V} = \vect{V_j} - \vect{V_i}$,
  correspondant à deux positions consécutives $i$ et $j$ de la balle.
\end{itemize}


\vressort{1}

\begin{enumerate}
\item \'Enoncer la deuxième loi de Newton.
\item Effectuer l'inventaire des forces agissant sur la balle de golf.
\item Indiquer la conséquence de la deuxième loi de Newton et comparer
  avec les faits expérimentaux observés.
\end{enumerate}
