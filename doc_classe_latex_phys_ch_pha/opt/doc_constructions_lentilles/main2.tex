\section*{Constructions géométriques pour les lentilles minces convergentes}
\vspace{-0.5cm}

\[f' = \overline{OF'} > 0\]

\vressort{1}

%\medskip


\noindent
\optlcvdv{Cas 1}{<->}{objet réel}{$\overline{OA} \in ]-\infty ; 2f[$}
\optlcvdv{Cas 2}{<->}{objet réel}{$\overline{OA} \in ]2f ; f[$}
\optlcvdv{Cas 3}{<->}{objet réel dans le plan focal objet}{$\overline{OA} = f$}
\optlcvdv{Cas 4}{<->}{objet réel entre le plan focal objet et la lentille}{$\overline{OA} \in ]f ; 0[$}
\optlcvdv{Cas 5}{<->}{objet virtuel}{$\overline{OA} \in ]0 ; \infty[$}
\optlcvdv{Cas 6}{<->}{objet à l'infini, réel ou virtuel}{$\overline{OA} = \pm \infty$}
