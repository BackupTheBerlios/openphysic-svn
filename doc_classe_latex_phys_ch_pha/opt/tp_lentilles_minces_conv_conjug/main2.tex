\tp[Relations de conjugaison et de grandissement des lentilles minces convergentes]
{Relations de conjugaison\\
% et de grandissement\\
des lentilles minces convergentes}

\nomprenomclasse


\begin{multicols}{2}

\objectifs{
 \item \'Etablir la relation de conjugaison des lentilles
   minces (vérifiées expérimentalement dans le cas des lentilles minces
   convergentes mais vérifiables aussi dans le cas des lentilles minces
   divergentes)
 \item \'Etablir la relation de grandissement des lentilles minces.
}

\bigskip

\materiel{
 \item Banc optique
 \item Objet lumineux (lanterne avec lettre $F$)
 \item \'Ecran millimétré.
 \item Lentille mince convergente de vergence  \[V = \troufixe{1.5cm}\]
}

\end{multicols}

\section{Interprétation}

\begin{enumerate}

%  \item Faire la construction graphique pour quatre positions différentes.

% \begin{itemize}
% \item $\ma{AO} > 2f'$
% \item $f < \ma{AO} < 2f'$

\item Faire la construction graphique à l'échelle pour deux positions différentes.\\
Indiquer dans chaque cas les caractéristiques de l'image.

% \item Récapituler les 4 cas possibles de position d'un objet par rapport au centre d'une lentille

% \item Calculer $\displaystyle \frac{1}{\ma{OA'}}$,
%   $\displaystyle \frac{1}{\ma{OA}}$ et $\displaystyle \frac{1}{\ma{OF'}}$ et noter les valeurs dans le tableau.

 \item Pour chaque position, calculer
  $\displaystyle \frac{1}{\ma{OA}}$,
  $\displaystyle \frac{1}{\ma{OA'}}$,
  $\displaystyle \frac{1}{\ma{OA'}} - \frac{1}{\ma{OA}}$, 
%   $\displaystyle \frac{\ma{OA'}}{\ma{OA}}$,
%   $\displaystyle \frac{\ma{A'B'}}{\ma{AB}}$.
 et noter les valeurs dans le tableau.

 \item Réaliser une représentation graphique de $\displaystyle y =
   \frac{1}{\ma{OA'}}$ en fonction de $\displaystyle x =
   \frac{1}{\ma{OA}}$.

 \item Quelle relation la représentation graphique permet-elle
   d'établir ? Prolonger la courbe jusqu'à l'axe des ordonnées. Que
   représente ce point ? Comparer son ordonnée à la vergence $V$ de la
   lentille.

 \item \'Etablir une relation entre $\displaystyle
   \frac{1}{\ma{OA'}}$, $\displaystyle \frac{1}{\ma{OA}}$
   et $\displaystyle \frac{1}{\ma{OF'}}$.

 \item \`A l'aide de ces résultats, décrire une méthode de mesure de
   la distance focale image d'une lentille de vergence $V$ inconnue.

 \item Pour chaque position, calculer le rapport
   $\displaystyle \frac{\ma{OA'}}{\ma{OA}}$
   et noter les valeurs dans le tableau.
% le comparer au rapport
%   $\displaystyle \frac{\ma{A'B'}}{\ma{AB}}$.

 \item Pour chaque position, calculer le rapport
   $\displaystyle \frac{\ma{A'B'}}{\ma{AB}}$
   et noter les valeurs dans le tableau.

 \item D'après votre tableau de mesure, quelle grandeur est égale au
   grandissement $\displaystyle \gamma = \frac{\ma{A'B'}}{\ma{AB}}$.



\end{enumerate}
