\section{Réfraction}

\subsection{Réfraction air-plexiglas}

\begin{center}
\begin{pspicture}(-7,-2)(7,2)
%\psgrid[subgriddiv=1,griddots=10]
%\pnode(6,0){A}
\optdisquegradue

% demi-cylindre coté plan
\pswedge[fillstyle=vlines](0,0){0.75}{270}{90}


% rayon incident
\rput(-5.5,-2){rayon incident}
\psline(-4,-2)(0,0) \psline{->}(-4,-2)(-2,-1)
\rput(-3.5,-0.5){$i_1$ : angle d'incidence}
\pswedge(0,0){0.75}{180}{206.6}

% rayon réfracté
\rput(5.5,1){rayon réfracté}
\psline(0,0)(4,1) \psline{>-}(2,0.5)(4,1)
\rput(3.5,0.25){$i_2$ : angle de réfraction}
\pswedge(0,0){1}{0}{14}


% légende milieu
\rput(-5,2){AIR - milieu transparent n° 1}
\rput(5,2){PLEXIGLAS - milieu transparent n° 2}
\psline{->}(2,1.75)(0.25,0.25)

\end{pspicture}
\end{center}


\begin{enumerate}
\item Remplacer le miroir par un demi-cylindre en plexiglas.

\item Envoyer le pinceau lumineux en direction de la face plane du demi cylindre.

\item Faire varier l'angle d'incidence $i$ (par rapport à la normale au plan du demi-cylindre). Mesurer l'angle de réflexion $r$. Reporter les résultats expérimentaux dans le tableau suivant.

\begin{arraydata}{10}
\hline
Angle d'incidence $i_1$ ($°$)  & 0 & 10 & 20 & 30 & 40 & 50 & 60 & 70 & 80 & 85 \\ \hline
\rule[-1cm]{0cm}{2cm}%
Angle de réfraction $i_2$ ($°$) &   &    &    &    &    &    &    &    &    &    \\ \hline
\rule[-1cm]{0cm}{2cm}%
$\displaystyle \frac{sin(i_1)}{sin(i_2)}$    &   &    &    &    &    &    &    &    &    &    \\ \hline
\end{arraydata}
%\frac{sin(i_1)}{sin(i_2}}

\item Conclusion.

\bigskip

\item Calculer l'indice de réfraction $n_2$ du plexiglas.

\item Mesurer puis calculer l'angle d'incidence limite ${i_1}_{lim}$ (s'il existe).

\item Mesurer puis calculer l'angle de réfraction limite ${i_2}_{lim}$ (s'il existe).
\end{enumerate}
