\objectifs{
\item Préparer une solution électrolytique en mettant en solution un
  solide ionique.
\item \'Ecrire une équation de mise en solution.
\item Distinguer la concentration molaire de soluté apporté et les
  concentrations effectives des ions en solution.
\item Identifier des ions en solution à l'aide de tests par
  précipitation.
\item \'Ecrire une équation de précipitation.
}

\section{Préparation de la solution de sulfate de fer (III)}


\begin{enumerate}
\item Préparer $100~mL$ d'une solution de sulfate de fer (III) de
  concentration $c = 3,5.10^{-3}~mol.L^{-1}$ à partir du solide
  $Fe_2(SO_4)_3$.

\item Indiquer sur le compte rendu à l'aide de schéma le protocole
  expérimental pour réaliser cette dissolution.

\item \'Ecrire l'équation bilan de la dissolution du cristal
  ionique. En déduire les concentrations effectives des ions présents
  en solution.

\end{enumerate}

\section{Vérification expérimentale des quantités d'ions fer (III) et d'ions sulfate}


\begin{enumerate}
\item Prélever deux fois $20~mL$ de la solution de sulfate de fer que
  l'on placera dans 2 béchers distincts ($1$ et $2$).

\item Calculer le nombre de mol d'ions $Fe^{3+}$ et d'ions sulfate
  $SO_4^{2-}$ présents dans ces $20~mL$.

\item On donne les réactions  de précipitations suivantes : 

 \begin{itemize}
 \item les ions $Fe^{3+}$ réagissent avec les ions hydroxydes $HO^{-}$ en
donnant un précipité marron

 \item les ions $SO_4^{2-}$ réagissent avec les ions baryum $Ba^{2+}$ en
donnant un précipité blanc.
 \end{itemize}

Ecrire les équations bilan des deux réactions de précipitations.

\item Calculer les quantités de matière à utiliser en ions $HO^{-}$ et
  en ions $Ba^{2+}$ pour faire complètement réagir par précipitations
  les ions contenus dans les $20~mL$ de la solution de sulfate de fer
  III.

\item Calculer les volumes d'hydroxyde de sodium ($Na^{+} + HO^{-}$)
  et de chlorure de baryum ($Ba^{2+} + 2Cl^{-}$) pour faire
  complètement réagir les ions $Fe^{3+}$ et $SO_4^{2-}$.

\item Placer la solution de soude ($Na^+$ + $HO^-$) dans une burette
  puis réaliser la précipitation des ions $Fe^{3+}$ dans le bécher
  noté $1$ en vous plaçant dans les conditions expérimentales de la
  question $5$.


\item Rincer correctement la burette puis y placer la solution de
  chlorure de baryum ($Ba^{2+} + 2Cl^-$). Réaliser la précipitation
  des ions sulfates $SO_4^{2-}$ dans le bécher noté $2$ en vous
  plaçant dans les conditions expérimentales de la question $5$.

\item Réaliser des tests complémentaires sur les produits de vos
  réactions de précipitations permettant de confirmer l'absence d'ions
  $Fe^{3+}$ et $SO_4^{2-}$.

\item Conclure sur la validité de vos calculs théoriques.
\end{enumerate}
