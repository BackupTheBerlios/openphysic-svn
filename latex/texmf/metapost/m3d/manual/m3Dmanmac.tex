\def\@chapter#1{%

	\ifnormalchapter
		\edef\lastref{\uppercase
			\expandafter{\romannumeral\c@chapter}}%

		\mark{{\@titlerunning}{\currentchapter}}%

	\else
		\edef\lastref{#1}%
		\let\currentchapter=\lastref
		\edef\titlemark{{\@titlerunning}{}}%
		\mark{{\@titlerunning}{}}%
